// --- SUSTITUYE ESTA FUNCIÓN EN TU CÓDIGO --- async function handleAuthSubmit() { const email = document.getElementById('auth-email').value; const pass = document.getElementById('auth-pass').value; if(!email || !pass) return alert("Por favor, completa los datos."); const btn = document.getElementById('btn-main-auth'); const originalText = btn.innerText; btn.innerHTML = `
`; btn.disabled = true; // Mutación para obtener el token de acceso del cliente (Login) const loginMutation = ` mutation { customerAccessTokenCreate(input: {email: "${email}", password: "${pass}"}) { customerAccessToken { accessToken expiresAt } customerUserErrors { code field message } } } `; try { const response = await fetch(`https://${CONFIG.shop}/api/2024-01/graphql.json`, { method: 'POST', headers: { 'Content-Type': 'application/json', 'X-Shopify-Storefront-Access-Token': CONFIG.token }, body: JSON.stringify({ query: loginMutation }) }); const res = await response.json(); const authData = res.data.customerAccessTokenCreate; if (authData.customerAccessToken) { // ÉXITO: Guardamos el token para que la app reconozca al cliente const token = authData.customerAccessToken.accessToken; localStorage.setItem('k_token', token); localStorage.setItem('k_email', email); // Ahora obtenemos el nombre real del cliente para la Wallet await fetchCustomerName(token); // Cerramos el panel y actualizamos la UI sin salir de la app toggleProfile(); checkSession(); alert("¡Bienvenido al Supermercado KOLMARD!"); } else { // ERROR: Credenciales incorrectas const errorMsg = authData.customerUserErrors[0]?.message || "Error al iniciar sesión"; alert(errorMsg); btn.innerText = originalText; btn.disabled = false; } } catch (e) { alert("Error de conexión. Inténtalo de nuevo."); btn.innerText = originalText; btn.disabled = false; } } // --- FUNCIÓN PARA TRAER EL NOMBRE REAL DEL CLIENTE --- async function fetchCustomerName(token) { const customerQuery = ` query { customer(customerAccessToken: "${token}") { firstName lastName } } `; try { const response = await fetch(`https://${CONFIG.shop}/api/2024-01/graphql.json`, { method: 'POST', headers: { 'Content-Type': 'application/json', 'X-Shopify-Storefront-Access-Token': CONFIG.token }, body: JSON.stringify({ query: customerQuery }) }); const res = await response.json(); if(res.data.customer) { const name = res.data.customer.firstName; localStorage.setItem('k_name', name); document.getElementById('wallet-name').innerText = name; } } catch (e) { console.error("Error al obtener nombre"); } }
